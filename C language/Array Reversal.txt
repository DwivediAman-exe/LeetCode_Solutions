#include <stdio.h>
#include <stdlib.h>

int main()
{
    int n,i;
    scanf("%d",&n);
    int *arr=(int*)malloc(n*sizeof(int));
    for(int i=0;i<n;i++)
        scanf("%d",&arr[i]);

    /* Write the logic to reverse the array. */

    for(i = n-1; i >=0; i--)
        printf("%d ", arr[i]);
    return 0;
}
